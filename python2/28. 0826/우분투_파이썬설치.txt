# 파이썬 설치
sudo apt install python3 python3-pip -y
sudo apt install python3-dev python3-venv -y

# 프로젝트 폴더 생성
mkdir ~/flask_app
cd ~/flask_app

# 가상환경 설치
python3 -m venv venv

# 가상환경 활성화
source venv/bin/activate

# 플라스크 설치
pip install flask

# MySQL 모듈 설치
pip install mysql-connector-python

# 파이썬 프로젝트 복사
flask_login 폴더를 ~/flask_app 안으로

# 하위 프로젝트 이동
cd ~/flask_app/flask_login

# flask 실행
python app.py
